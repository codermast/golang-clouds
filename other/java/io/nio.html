    public static void main(String[] args) throws IOException {
        // 1. åˆ›å»º Selector
        Selector selector = Selector.open();
        
        // 2. åˆ›å»º ServerSocketChannel
        ServerSocketChannel serverChannel = ServerSocketChannel.open();
        serverChannel.bind(new InetSocketAddress(8080));
        serverChannel.configureBlocking(false);  // éé˜»å¡
        
        // 3. æ³¨å†Œåˆ° Selector
        serverChannel.register(selector, SelectionKey.OP_ACCEPT);
        System.out.println("æœåŠ¡å™¨å¯åŠ¨ï¼Œç›‘å¬ç«¯å£ 8080");

                }

        ServerSocketChannel serverChannel = (ServerSocketChannel) key.channel();

        clientChannel.register(selector, SelectionKey.OP_READ);
        System.out.println("å®¢æˆ·ç«¯è¿æ¥ï¼š" + clientChannel.getRemoteAddress());
    }
    
    static void handleRead(SelectionKey key) throws IOException {

        ByteBuffer buffer = ByteBuffer.allocate(1024);
        
        int bytesRead = channel.read(buffer);
        if (bytesRead > 0) {

            channel.close();
        }
    }
