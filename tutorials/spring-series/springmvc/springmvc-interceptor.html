@Component
public class TokenInterceptor implements HandlerInterceptor {
    
    @Autowired
    private TokenService tokenService;
    
    @Override
    public boolean preHandle(HttpServletRequest request, 
                            HttpServletResponse response, 
                            Object handler) throws Exception {
        // è·å– Token
        String token = request.getHeader("Authorization");
        
        if (token == null || token.isEmpty()) {
            sendError(response, 401, "ç¼ºå°‘ Token");
            return false;

        // éªŒè¯ Token
        if (!tokenService.verify(token)) {
            sendError(response, 401, "Token æ— æ•ˆæˆ–å·²è¿‡æœŸ");
            return false;

        // å°†ç”¨æˆ·ä¿¡æ¯æ”¾å…¥è¯·æ±‚å±æ€§

        return true;
    }
    
    private void sendError(HttpServletResponse response, int code, String message) 
            throws IOException {
        response.setContentType("application/json;charset=UTF-8");
        response.setStatus(code);
        response.getWriter().write(
            String.format("{\"code\":%d,\"message\":\"%s\"}", code, message)
        );
    }
}

@Component
@Slf4j
public class LogInterceptor implements HandlerInterceptor {
    
    private static final String START_TIME = "requestStartTime";
    
    @Override
    public boolean preHandle(HttpServletRequest request, 
                            HttpServletResponse response, 
                            Object handler) throws Exception {
        // è®°å½•å¼€å§‹æ—¶é—´
        request.setAttribute(START_TIME, System.currentTimeMillis());
        
        // è®°å½•è¯·æ±‚ä¿¡æ¯
        log.info("è¯·æ±‚å¼€å§‹ - URI: {}, Method: {}, IP: {}", 
                request.getRequestURI(),
                request.getMethod(),
                getClientIP(request));
        
        return true;
    }
    
    @Override
    public void afterCompletion(HttpServletRequest request, 
                               HttpServletResponse response, 
                               Object handler, 
                               Exception ex) throws Exception {
        // è®¡ç®—è€—æ—¶
        Long startTime = (Long) request.getAttribute(START_TIME);
        long duration = System.currentTimeMillis() - startTime;
        
        // è®°å½•å“åº”ä¿¡æ¯
        log.info("è¯·æ±‚ç»“æŸ - URI: {}, Status: {}, è€—æ—¶: {}ms", 
                request.getRequestURI(),
                response.getStatus(),
                duration);
        
        // è®°å½•å¼‚å¸¸
        if (ex != null) {
            log.error("è¯·æ±‚å¼‚å¸¸ - URI: {}, Error: {}", 
                    request.getRequestURI(), ex.getMessage());
        }
    }
    
    private String getClientIP(HttpServletRequest request) {
        String ip = request.getHeader("X-Forwarded-For");
        if (ip == null || ip.isEmpty() || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getHeader("X-Real-IP");
        }
        if (ip == null || ip.isEmpty() || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getRemoteAddr();
        }
        return ip;
    }
}
